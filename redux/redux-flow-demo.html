// 1. Action Types
const DEPOSIT  = 'account/deposit';
const WITHDRAW = 'account/withdraw';

// 2. Action Creators
const deposit  = (amount) => ({ type: DEPOSIT,  payload: amount });
const withdraw = (amount) => ({ type: WITHDRAW, payload: amount });

// 3. Reducer (pure function: (state, action) => newState)
function accountReducer(state = { balance: 0 }, action) {
  switch (action.type) {
    case DEPOSIT:  return { ...state, balance: state.balance + action.payload };
    case WITHDRAW: return { ...state, balance: state.balance - action.payload };
    default:       return state;
  }
}

// 4. Store (holds state, accepts dispatches, notifies subscribers)
function createStore(reducer) {
  let state = reducer(undefined, { type: '@@INIT' });
  const listeners = [];
  return {
    getState:   () => state,
    dispatch:   (action) => { state = reducer(state, action); listeners.forEach(fn => fn()); },
    subscribe:  (fn) => { listeners.push(fn); return () => listeners.splice(listeners.indexOf(fn), 1); },
  };
}

// 5. Wire it up
const store = createStore(accountReducer);
store.subscribe(() => render(store.getState()));   // UI re-renders on state change
depositBtn.onclick  = () => store.dispatch(deposit(10));
withdrawBtn.onclick = () => store.dispatch(withdraw(10));
